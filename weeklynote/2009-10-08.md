     2 for (( x=1; \$x < 30; x++ ))

     4 
     5 DISPLAY=:0 osd_cat \\
     6   --pos=middle \\
     7   --lines=7 \\
     8   --align=center \\
     9   --font='10x20' \\
    10   --color=green \\
    11   --delay=1 <<EOF 
    12 $(date +%H:%M)
    13 
    14 Take a rest...
    15 
    16 $(date +%H:%M)
    17 EOF
    18 
    19 sleep 1
    20 
    21 done
    22 " | at now + 50min
